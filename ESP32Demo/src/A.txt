#include <Arduino.h>

const int rxPin = 3; // 替换为您要使用的GPIO引脚

void setup() {
  pinMode(rxPin, INPUT);
  Serial.begin(115200);
}

void loop() {
  Serial.println(digitalRead(rxPin));
}

#include <Arduino.h>

void setup()
{
  Serial.begin(9600);

  pinMode(2, OUTPUT);
}

void loop()
{
  if (Serial.read() == 1)
  {
    digitalWrite(2, LOW);
  }
  else
  {
    digitalWrite(2, HIGH);
  }
}
